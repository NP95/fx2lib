#define RESETTOGGLE (   ep)
Value:
do {\
        BYTE x = ep; \
        if (x&0x80) { x |= 0x10; } \
        x &= 0x1F; \
        TOGCTL = x; \
        TOGCTL = x | bmRESETTOGGLE; \
} while (0)
